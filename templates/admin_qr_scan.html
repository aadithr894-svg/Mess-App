const resultDiv = document.getElementById("result"); const liveCount = document.getElementById("liveCount"); const mealTypeSpan = document.getElementById("mealType"); let lastScanTime = 0; // Auto-detect meal based on current time function detectMeal() { const now = new Date(); const hours = now.getHours() + now.getMinutes()/60; if (hours >= 7.5 && hours <= 9) return 'breakfast'; if (hours >= 12 && hours <= 14) return 'lunch'; return 'dinner'; } // Update live count periodically (every 2 minutes) function updateLiveCount() { const meal = detectMeal(); mealTypeSpan.innerText = meal; fetch(`/admin/live_count/${meal}`) .then(res => res.json()) .then(data => { if (data.success) liveCount.innerText = data.count; }) .catch(()=>{ liveCount.innerText = "0"; }); } // Run once on load and then every 2 minutes updateLiveCount(); setInterval(updateLiveCount, 120000); // 120000ms = 2 minutes // Handle successful QR scan function onScanSuccess(decodedText) { const now = Date.now(); if (now - lastScanTime < 500) return; // throttle 0.5s lastScanTime = now; let user_id = null; try { const parts = decodedText.split(","); for (let p of parts) if (p.startsWith("user_id:")) user_id = p.split(":")[1]; } catch(e){} if (!user_id) { resultDiv.textContent = "❌ Invalid QR code!"; resultDiv.className = "text-center text-red-600 font-semibold mb-4"; return; } const meal = detectMeal(); fetch("/admin/scan_qr", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ user_id, meal_type: meal }) }) .then(res => res.json()) .then(data => { if (data.success) { resultDiv.textContent = `✅ ${data.name} marked for ${meal}`; resultDiv.className = "text-center text-green-600 font-semibold mb-4"; // DON'T update live count here to reduce load } else { resultDiv.textContent = `❌ ${data.message}`; resultDiv.className = "text-center text-red-600 font-semibold mb-4"; } }) .catch(() => { resultDiv.textContent = "❌ Network error!"; resultDiv.className = "text-center text-red-600 font-semibold mb-4"; }); } // QR scanner const html5QrcodeScanner = new Html5QrcodeScanner("reader", { fps: 15, qrbox: 250, experimentalFeatures: { useBarCodeDetectorIfSupported: true } }); html5QrcodeScanner.render(onScanSuccess, () => {});
